4、SpringMVC RestFul风格.

RestfulController.

@Controller
public class RestfulController {

    //get请求方式获取的数据
    //@PathVariable : 把方法形参绑定到url模板变量上，（名要和{}里的相同），产生了restful的风格的url
    @RequestMapping(value="/restful/{param}",method = RequestMethod.GET)
    public String restful(@PathVariable int param, Model model){
        String content = "get请求传来的内容是" + param;
        model.addAttribute("msg",content);
        return "test";
    }

    //post请求方式获取数据
    @PostMapping(value = "/restful/{param}")
    public String restful(@PathVariable String param, Model model){
        String content = "post请求传来的内容是" + param;
        model.addAttribute("msg",content);
        return "test";
    }
}

    <!--
    前面 springMVC 2 3
    是 springmvc实现的两种方式
    这里讲一下RestFul风格
    以往我们 发起请求主要是 表单(/addUser.action)和地址栏(?xxx=xx)
    通过不同的参数来实现不同的效果（不同的请求方式，使用的url不同，主要通过属性method达到不同的效果）

    Restful就是一个资源定位及资源操作的风格。不是标准也不是协议，只是一种风格。基于这个风格
设计的软件可以更简洁，更有层次，更易于实现缓存等机制。
    Restful风格请求的url： /item/1/2/
    好处：1.更加简洁
         2.更加安全
         3.根据不同的请求方式来实现不同的效果（请求地址（url）可以一样，但是不同的请求方式实现的功能可以不同）
    -->

    }

        this.bookMapper = bookMapper;
    }

    @Override
    public int addBook(Books books) {
        return bookMapper.addBook(books);
    }

    @Override
    public int deleteBookById(int id) {
        return bookMapper.deleteBookById(id);
    }

    @Override
    public int updateBook(Books books) {
        return bookMapper.updateBook(books);
    }

    @Override
    //分页查询 map: bookID,bookName,bookCounts,detail    只需管currentPage
    public Page queryBooks(@Nullable Map<String, Object> map) {

        if (map == null) {
            map = new HashMap<>();
        }
        Page page = new Page();
        //limit start,size
        Integer currentPage = (Integer) map.get("currentPage");

        //page导入count可以计算出 ==> pageCount(总页数)的值
        page.setCount(bookMapper.getCount(map));
        //如果当前页为null，内部 使用默认页：1 如果不为null，内部自动设置为 该值
        //page导入currentPage 可计算出 ==> start(当前页第一条记录的偏移量)的值
        //这一条语句 是用于重置
        page.setCurrentPage(currentPage);
        //
        map.put("start",page.getStart());
        map.put("size",page.getPageSize());
        //设置当前页列表
        page.setList(bookMapper.queryBooks(map));

        return page;
    }
}

Page.

/*
    调用时，要设置两东西 currentPage(可选) 和 count 和 list(必须)

    pageSize 是固定的
    setCount  ==> setPageCount ( pageCount = count / pageSize )
    setCurrentPage ==> setOffset( offset = (current - 1) * pageSize )

    list 主要来源是 查询数据库获取的记录
    currentPage 是由前端传入，currentPage的初始值设为 1，offset初始值刚好为 0; 不设置就是默认首页
    count 是 查询出来的总记录数

    顺序有严格的规定(setPageSize) ==> setCount ==> setCurrentPage ==> 使用Page的getStart，getPageSize 完成 sql分页
    ==> 得到的结果 设置会page中 setList ==> 将Page返回个前端
 */
public class Page {
    //总页数
    private Integer pageCount = 0;
    //记录数
    private Integer count = 0;
    //页容量 初始容量 为 7
    private Integer pageSize = 7;
    //当前页 初始为首页
    private Integer currentPage = 1;
    //当前页第一条记录的偏移量
    private Integer start = 0;
    //查询到的记录
    private List<?> list;


    public Page() {
    }

    public List<?> getList() {
        return list;
    }

    public void setList(List list) {
        this.list = list;
    }

    public Integer getStart() {
        return start;
    }

    //设置好当前页时 会调用
    private void setStart() {
        if (this.currentPage != 0){
            this.start = (this.currentPage - 1) * this.pageSize;
        }//否则 start 使用默认 0
    }

    public Integer getPageCount() {
        return pageCount;
    }

    //在设置count数后 会调用
    private void setPageCount() {
        this.pageCount = (int) Math.ceil(this.count * 1.0 / this.pageSize);
    }

    public Integer getCount() {
        return count;
    }

    public void setCount(Integer count) {
        if (count != null) {
            this.count = count;
        }//否则 count = 0; pageCount = 0;
        if (count == 0){// 记录数为 0 设置 currentPage = 0 ;
            setCurrentPage(0);
        }
        setPageCount();
    }

    public Integer getPageSize() {
        return pageSize;
    }

    public void setPageSize(Integer pageSize){

        if (pageSize != null && pageSize > 5) {
            this.pageSize = pageSize;
        }//否则 pageSize = 7 ==> 每页7条记录
    }

    public Integer getCurrentPage() {
        return currentPage;
    }

    public void setCurrentPage(Integer currentPage) {
        if (currentPage != null){
            if(currentPage < 1){
               this.currentPage = 1;
            }else if(currentPage > this.pageCount){
                this.currentPage = this.pageCount;
            }else {
                this.currentPage = currentPage;
            }
        }//否则 currentPage = 1 ; start = 0 ==> 第一页
        if (count == 0){
            this.currentPage = 0;
        }
        setStart();
    }

    @Override
    public String toString() {
        return "Page{" +
                "pageCount=" + pageCount +
                ", count=" + count +
                ", pageSize=" + pageSize +
                ", currentPage=" + currentPage +
                ", start=" + start +
                ", list=" + list +
                '}';
    }

}
